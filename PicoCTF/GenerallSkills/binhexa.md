def bin_operations(a, b, operations):

  if operations == '+':

    result = bin(a + b)

  elif operations == '-':

    result = bin(a - b)

  elif operations == '*':

    result = bin(a * b)

  elif operations == '>>':

    num = input("Choose number a or b: ")

    if num == 'a':

      result = bin(a >> 1)

    elif num == 'b':

      result = bin(b >> 1)

    else:

      print("Invalid number to choose")

  elif operations == '<<':

    num = input("Choose number a or b: ")

    if num == 'a':

      result = bin(a << 1)

    elif num == 'b':

      result = bin(b << 1)

    else:

      print("Invalid number to choose")

  elif operations == '&':

    result = bin(a & b)

  elif operations == '|':

    result = bin(a | b)

  else:

    print("Invalid operations")

  return(result)

  

if __name__ == "__main__":
